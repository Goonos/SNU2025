XMLHttpRequest let xhr = new XMLHttpRequest(); – XMLHttpRequest 객체 인스턴스를 생성해 xhr 변수에 할당 xhr.open("GET", "student.json"); – GET 방식으로 student.json 리소스에 대한 요청을 초기화 xhr.send(); – 설정된 요청을 서버에 전송 2. JSON 자료 가져와 표시하기 let xhr = new XMLHttpRequest(); – XMLHttpRequest 객체 인스턴스를 생성해 xhr 변수에 할당 xhr.open("GET", "student.json"); – GET 방식으로 student.json에 대한 요청을 초기화 xhr.send(); – 설정된 요청을 서버에 전송 . xhr.onreadystatechange = function () { – 요청 상태가 변경될 때 실행할 콜백 함수 등록 if (xhr.readyState === 4 && xhr.status === 200) { – 요청이 완료(4)되고 성공(200)했는지 검사 let student = JSON.parse(xhr.responseText); – 응답 텍스트를 JSON 객체로 파싱해 student 변수에 저장 document.getElementById("표사할영역").innerHTML = 백틱 – #.result 요소의 HTML 콘텐츠를 템플릿으로 설정 <.h1>${student.name}<./h1> <.ul> <.li>전공: ${student.major}<./li> <.li>학년: ${student.grade}<./li> <./ul> 백틱; – 템플릿 리터럴 끝 } – if 문 종료 }; – onreadystatechange 등록 블록 종료 3. 예외처리 - try catch finally try { – 예외 발생 가능성이 있는 코드 블록 시작 … – 실행할 코드(생략) } catch(err) { – 예외 발생 시 err 변수로 예외 객체 받아 처리 console.log(`오류 발생 : ${err}`); – 전체 예외 객체를 템플릿 문자열에 보간해 에러 로그 출력 } – catch 블록 종료 console.log("끝"); - try catch finally 빨간 오류메세지 try { – 예외 발생 가능성이 있는 코드 블록 시작 … – 실행할 코드(생략) } catch(err) { – 예외 발생 시 err 변수로 예외 객체 받아 처리 console.error(`오류 발생 : ${err}`); – 전체 예외 객체를 템플릿 문자열에 보간해 에러 로그 출력 console.error(`오류 발생 : ${err.name}`); – 예외 객체의 이름(err.name)을 보간해 에러 로그 출력 console.error(`오류 발생 : ${err.message}`); – 예외 객체의 메시지(err.message)를 보간해 에러 로그 출력 } – catch 블록 종료 - throw 에러 일부러 만들기 let json = '{"grade": 3, "age": 25}'; – grade와 age를 속성으로 갖는 JSON 문자열을 json에 할당 try { – 예외 발생 가능성이 있는 코드 블록 시작 let user = JSON.parse(json); – JSON 문자열을 객체로 변환해 user에 할당 if (!user.name) { – user.name이 없으면 throw "사용자 이름이 없습니다."; – 예외 메시지를 던짐 } – if 블록 종료 } catch (err) { – 예외 발생 시 err 객체로 catch 블록 진입 console.error(err); – err 값을 에러 레벨로 콘솔에 출력 } – catch 블록 종료
