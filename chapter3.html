*반복문

students = ['1', '2', '3', '4']

for(let i = 0; i < students.length; i++) {
  document.write(\`\${students[i]}\`)
}

for (변수 of 객체)를 사용할 경우 배열의 값을 순서대로 가져온다
for(let student of students) { document.write(\`\${student}\`); }

배열의 길이가 변할 땐 forEach 사용
students.forEach(function(student) {
  document.write(\`\${student}\`);
})

함수는 화살표 함수로 변경 가능
students.forEach(student => { document.write(\`\${student}\`); })

객체에서 for문을 쓸 땐 for in 사용

for (key in subject) { document.write(\`\${key} : \${subject[key]}\`); }

구구단
for(let i = 1; i <= 9; i++) {
  for(let j = 1; j <= 9; j++) {
    document.write(\`\${i} X \${j} = \${i*j}
\`);
  }
}
