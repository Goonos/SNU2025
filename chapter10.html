10장: 문자열 다루기 - 문자열 길이 확인 let str = "Hello"; str.length // 5 - 문자 추출 str.charAt(3) // 3번째 문자 str[3] // 배열처럼 접근 가능 - 특정 문자 개수 세기 function counting(str, ch) { let count = 0; for(let i = 0; i < str.length; i++) { if(str[i] === ch) count++; } return count; } - 문자열 검색 • indexOf("문자열"): 첫 등장 위치 반환, 없으면 -1 • lastIndexOf("문자열"): 마지막 등장 위치 반환 • includes("문자열"): 포함 여부 true/false 반환 • startsWith("문자열"), endsWith("문자열"): 시작/끝 문자 확인 - 문자열 자르기 • substring(start, end) • slice(start, end) — 음수 인덱스 가능 예) str.substring(5) // 5번째부터 끝까지 str.slice(-5) // 뒤에서 5글자부터 끝까지 - 문자열 분할(split) str.split("구분자") → 배열 반환 예) "Hello world".split(" ") → ["Hello", "world"] - 문자열 공백 제거 str.trim(), trimStart(), trimEnd() - 대소문자 변환 str.toUpperCase(), str.toLowerCase() - 이메일 처리 예시 • 이메일을 @ 기준으로 분리하여 사용자명과 도메인 추출 • 사용자명 일부만 보여주고 도메인은 그대로 표시하는 간단한 처리 • split(), substring() 사용 - 예) let email = "username@example.com"; let parts = email.split("@"); let user = parts[0].substring(0, 3); let domain = parts[1]; console.log(\`\${user}...@\${domain}\`);
